🧠 Step 2：Orchestrator（v2大脑）
📄 新建：
src/core/orchestrator/queryOrchestrator.ts
✨ 核心代码
import { buildContext } from "../context/contextBuilder";
import { runSQLAgent } from "../llm/sqlAgent";
import { validateSQL } from "../validator/sqlValidator";

export async function queryOrchestrator(userInput: string) {
  
  // 1️⃣ 构建上下文（schema + metrics）
  const context = await buildContext(userInput);

  // 2️⃣ 调用LLM生成SQL
  const sql = await runSQLAgent(userInput, context);

  // 3️⃣ 校验SQL（v2新增）
  const valid = validateSQL(sql, context.schema);

  if (!valid) {
    return "无法生成SQL";
  }

  return sql;
}

export async function buildContext(userInput: string) {
  
  const schema = loadSchema();
  const metrics = loadMetrics();

  // v2：这里可以扩展 query classifier
  return {
    schema,
    metrics,
    relations: schema.relations
  };
}

🤖 Step 6：LLM SQL Agent（核心推理层）
📄 新建：
src/core/llm/sqlAgent.ts
import OpenAI from "openai";

const client = new OpenAI({
  baseURL: "https://openrouter.ai/api/v1",
  apiKey: process.env.OPENROUTER_API_KEY!,
});

export async function runSQLAgent(userInput: string, context: any) {
  
  const completion = await client.chat.completions.create({
    model: "openai/gpt-4o-mini",
    messages: [
      {
        role: "system",
        content: `
你是企业级SQL Agent。

=== Schema ===
${JSON.stringify(context.schema, null, 2)}

=== Relations ===
${JSON.stringify(context.relations, null, 2)}

=== Metrics ===
${JSON.stringify(context.metrics, null, 2)}

规则：
1. 只能用schema字段
2. 必须遵守关系
3. 优先使用metrics
4. 不允许编造字段
5. 只输出SQL
        `
      },
      {
        role: "user",
        content: userInput
      }
    ]
  });

  return completion.choices[0].message.content;
}
🛡️ Step 7：SQL Validator（v2新增）
📄 新建：

export function validateSQL(sql: string, schema: any) {
  
  const tables = schema.tables.map((t: any) => t.name);

  // ❗简单v2版本校验
  const hasValidTable = tables.some((t: string) =>
    sql.includes(t)
  );

  return hasValidTable;
}
